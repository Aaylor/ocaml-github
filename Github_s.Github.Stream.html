val next : 'a t ->
       ('a * 'a t) option Github_s.Github.Monad.t
next s is the next element of the stream and a stream continuation if one exists.
val map : ('a -> 'b list Github_s.Github.Monad.t) ->
       'a t -> 'b t
map f s is the lazy stream of f applied to elements of s as they are demanded.
val find : ('a -> bool) ->
       'a t ->
       ('a * 'a t) option Github_s.Github.Monad.t
find p s is the first value in s satisfying p if one exists and a stream continuation.
val iter : ('a -> unit Github_s.Github.Monad.t) ->
       'a t -> unit Github_s.Github.Monad.t
iter f s is after the application of f to each element of s.
val to_list : 'a t -> 'a list Github_s.Github.Monad.t
to_list s is a list with each element of s.
val of_list : 'a list -> 'a t
of_list l is a stream with each element of l.
